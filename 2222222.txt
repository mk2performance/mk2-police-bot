import os
import asyncio
from flask import Flask, request
from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup
from telegram.ext import (
    ApplicationBuilder,
    ContextTypes,
    CallbackQueryHandler,
    MessageHandler,
    filters,
)

TOKEN = os.environ.get("TOKEN")
LOG_CHANNEL_ID = -1003880274713

pending_users = {}

app = Flask(__name__)
telegram_app = ApplicationBuilder().token(TOKEN).build()


async def new_member(update: Update, context: ContextTypes.DEFAULT_TYPE):
    for user in update.message.new_chat_members:
        if user.is_bot:
            return

        user_id = user.id
        chat_id = update.effective_chat.id

        pending_users[user_id] = chat_id

        await context.bot.restrict_chat_member(
            chat_id,
            user_id,
            permissions={}
        )

        keyboard = InlineKeyboardMarkup(
            [[InlineKeyboardButton("✅ Soy humano", callback_data=f"verify_{user_id}")]]
        )

        await context.bot.send_message(
            chat_id,
            f"🚔 Bienvenido {user.first_name}\n\n"
            "Pulsa el botón para verificar que eres humano.\n"
            "⏳ Tienes 5 minutos o serás expulsado.",
            reply_markup=keyboard
        )

        asyncio.create_task(check_user(context, chat_id, user_id))


async def check_user(context, chat_id, user_id):
    await asyncio.sleep(300)

    if user_id in pending_users:
        try:
            await context.bot.ban_chat_member(chat_id, user_id)
            await context.bot.send_message(
                LOG_CHANNEL_ID,
                f"❌ Usuario {user_id} expulsado por no verificar."
            )
        except:
            pass


async def verify_button(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    await query.answer()

    user_id = query.from_user.id

    if query.data == f"verify_{user_id}" and user_id in pending_users:
        chat_id = pending_users.pop(user_id)

        await context.bot.restrict_chat_member(
            chat_id,
            user_id,
            permissions={
                "can_send_messages": True,
                "can_send_media_messages": True,
                "can_send_other_messages": True,
                "can_add_web_page_previews": True
            }
        )

        await query.edit_message_text("✅ Verificación completada. Bienvenido.")

        await context.bot.send_message(
            LOG_CHANNEL_ID,
            f"✅ Usuario {user_id} verificado correctamente."
        )


telegram_app.add_handler(
    MessageHandler(filters.StatusUpdate.NEW_CHAT_MEMBERS, new_member)
)
telegram_app.add_handler(CallbackQueryHandler(verify_button))


@app.route(f"/{TOKEN}", methods=["POST"])
def webhook():
    update = Update.de_json(request.get_json(force=True), telegram_app.bot)
    asyncio.run(telegram_app.process_update(update))
    return "ok"


@app.route("/")
def home():
    return "MK2 Police Bot funcionando 🚔"


if __name__ == "__main__":
    PORT = int(os.environ.get("PORT", 10000))
    telegram_app.bot.set_webhook(
        url=f"https://{os.environ.get('RENDER_EXTERNAL_HOSTNAME')}/{TOKEN}"
    )
    app.run(host="0.0.0.0", port=PORT)
